
#决议用  必须设定 TIER  有 root, 授予 recipient = holder  
oe_create_titular_title_effect = {
	hidden_effect = {
		create_dynamic_title = {
			tier = $TIER$
			name = TITLE_NAME_FROM_TARGET
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		
		scope:new_title = {
			set_no_automatic_claims = yes
			set_can_be_named_after_dynasty = no
			set_color_from_title = root.capital_county
			set_capital_county = root.capital_county
			set_landless_title = yes
			change_title_holder = {
				holder = $HOLDER$
				change = scope:change
			}
			generate_coa = yes
			set_variable = {
				name = titular_title
				value = root.capital_county
			}
		}
		$HOLDER$ = {
			change_liege = {
				liege = root
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = scope:change

		scope:target = {
			reset_title_name = yes
		}
	
		$HOLDER$ = {
			set_primary_title_to = scope:new_title
		}
		$HOLDER$ = { #授予庄园
			if = {
				limit = {
					is_landed = no
					is_oe_aristocrat = no
				}
				domicile ?= {
					set_up_domicile_estate_effect = yes
				}
			}
		}
	}
}

#互动用
#仅针对 scope:recipient scope:target 注意修改
create_title_basing_target_effect = {
	hidden_effect = {
		scope:target = {
			change_title_name_$BASING_TIER$_effect = yes
		}
		
		create_dynamic_title = {
			tier = $NEW_TITLE_TIER$
			name = TITLE_NAME_FROM_TARGET
		}
		create_title_and_vassal_change = {
			type = created
			save_scope_as = change
			add_claim_on_loss = no
		}
		scope:new_title = {
			set_no_automatic_claims = yes
			set_can_be_named_after_dynasty = no
			set_color_from_title = scope:target
			set_capital_county = scope:target.title_capital_county
			set_landless_title = yes
			change_title_holder = {
				holder = scope:recipient
				change = scope:change
			}
		}
		scope:recipient = {
			change_liege = {
				liege = scope:actor
				change = scope:change
			}
		}

		resolve_title_and_vassal_change = scope:change

		scope:target = {
			reset_title_name = yes
		}
		scope:new_title = { generate_coa = yes }

		scope:recipient = {
			set_primary_title_to = scope:new_title
		}


		scope:new_title = {
			set_variable = {
				name = titular_title
				value = scope:target
			}
		}

		scope:target = {
			set_variable = {
				name = titular_granted
				value = scope:new_title
			}
		}
	}
}
