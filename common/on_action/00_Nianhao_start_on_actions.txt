	on_actions = {
		nianhao_pulse_on_action
	}

	effect = {
		every_ruler = {
			if = { #年号年份+1
				limit = {
					has_variable = nianhao_date
				}			
				change_variable = {
					name = nianhao_date
					add = 1
				}
			}
		}
	}
