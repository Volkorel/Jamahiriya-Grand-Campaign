am_china_1_list = {
	scope = character

	effect = {
		if = {
			limit = {
				root = {
					government_has_flag = government_is_celestial
				}				
			}
			root = {
				## Clear variable always before
				clear_variable_list = china_1_list
				every_courtier = {
					limit = {
						has_council_position = councillor_china_1
					}
					root = {
						add_to_variable_list = {
							name = china_1_list
							target = prev
						}
					}					
				}
				every_vassal = {
					limit = {
						has_council_position = councillor_china_1
					}
					root = {
						add_to_variable_list = {
							name = china_1_list
							target = prev
						}
					}					
				}
			}
		}
		if = {
			limit = {
				root = {
					NOT = {
						government_has_flag = government_is_celestial
					}	
					any_liege_or_above = {
						government_has_flag = government_is_celestial
					}				
				}
			}
			root = { clear_variable_list = china_1_list }
			root.liege = {
				every_courtier = {
					limit = {
						has_council_position = councillor_china_1
					}
					root = {
						add_to_variable_list = {
							name = china_1_list
							target = prev
						}
					}					
				}
				every_vassal = {
					limit = {
						has_council_position = councillor_china_1
					}
					root = {
						add_to_variable_list = {
							name = china_1_list
							target = prev
						}
					}					
				}
			}
		}
	}
}
