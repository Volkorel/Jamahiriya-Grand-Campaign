am_neishijian_list = {
	scope = character

	effect = {
		if = {
			limit = {
				root = {
					government_has_flag = government_is_celestial
				}				
			}
			root = {
				## Clear variable always before
				clear_variable_list = neishijian_list
				every_courtier = {
					limit = {
						has_council_position = councillor_neishijian
					}
					root = {
						add_to_variable_list = {
							name = neishijian_list
							target = prev
						}
					}					
				}
				every_vassal = {
					limit = {
						has_council_position = councillor_neishijian
					}
					root = {
						add_to_variable_list = {
							name = neishijian_list
							target = prev
						}
					}					
				}
			}
		}
		if = {
			limit = {
				root = {
					NOT = {
						government_has_flag = government_is_celestial
					}	
					any_liege_or_above = {
						government_has_flag = government_is_celestial
					}				
				}
			}
			root = { clear_variable_list = neishijian_list }
			root.liege = {
				every_courtier = {
					limit = {
						has_council_position = councillor_neishijian
					}
					root = {
						add_to_variable_list = {
							name = neishijian_list
							target = prev
						}
					}					
				}
				every_vassal = {
					limit = {
						has_council_position = councillor_neishijian
					}
					root = {
						add_to_variable_list = {
							name = neishijian_list
							target = prev
						}
					}					
				}
			}
		}
	}
}

show_china_offices_button = {
	scope = character
	is_shown = {
		OR = {
			AND = {
				exists = capital_province
				capital_province = { geographical_region = world_asia }
				highest_held_title_tier = tier_empire
			}
			any_liege_or_above = {
				exists = capital_province
				capital_province = { geographical_region = world_asia }
				highest_held_title_tier = tier_empire
			}					
		}
	}
}
