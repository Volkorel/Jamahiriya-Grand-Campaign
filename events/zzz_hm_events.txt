namespace = HM

HM.0051 = {
	type = character_event
	title = HM.0051.t
	theme = default
	#sender = scope:invited_character
	desc = {
		desc = HM.0051.desc
	}
	left_portrait = scope:great_misit
	override_background = {
		reference = throne_room
	}
	
	trigger = {
	}
	immediate = {
		save_scope_as = great_misit
		play_music_cue_once = { TRACK = Deus_Non_Vult }
		add_character_flag = xibixibi
	}
	option = {
		name = HM.0051.a
		trigger = {
			liege = {
				NOT = { 
					any_vassal = { 
							is_powerful_vassal = yes
							is_ai = yes
							sub_realm_size >= 10
						} 
				
				}	
			}
		}
		liege = {
			save_scope_as = great_misit_liege
		}
		if = {
			limit = {
				NOT = { has_variable = misit_power }
			}
			set_variable = {
				name = misit_power
				value = 0
			}
		}
		trigger_event = {
			on_action = great_misit_ongoing
			days = { 720 1280 }
		}
		add_trait = great_misit
		#great_misit_chronicle_effect = yes
		set_relation_great_misit_liege = liege
		
		liege = {
			set_relation_great_misit = root
			if = {
				limit = {
					NOT = { has_variable = misit_liege_power }
				}
				set_variable = {
						name = misit_liege_power
						value = 0
				}
			}
		}
		if = {
			limit = {
				NOT = { has_variable = misit_power }
			}
			set_variable = {
					name = misit_power
					value = 0
			}
		}
		misit_power_value_effect = yes
	}
	option = {
		name = HM.0051.b
		trigger = {
			liege = {
					any_vassal = { 
							is_powerful_vassal = yes
							is_ai = yes
							sub_realm_size >= 10
							is_adult = yes
					} 
					
			}
		}
		save_scope_as = baoansi
		liege = {
			save_scope_as = cuidaixing
			random_vassal = { 
				limit = { 
					is_powerful_vassal = yes
					sub_realm_size >= 10
					is_adult = yes
					is_ai = yes
				}
				save_scope_as = shoujingsi
			}
		}
		trigger_event = {
			id = great_misit.0034
			days = 3
		}
	}
}

HM.0053 = {
	type = character_event
	hidden = yes
	trigger = {
		has_variable = misit_power
		#is_landed = yes
	}
	immediate = {
		if = {
			limit = { exists = designated_heir }
			designated_heir = {
				if = {
					limit = {
						NOT = { has_variable = misit_power }
					}
					set_variable = {
						name = misit_power
						value = 0
					}
				}
				add_trait = great_misit
				#great_misit_chronicle_effect = yes
				trigger_event = {
					id = HM.0057
					days = 3
				}
				set_relation_great_misit_liege = root.liege
				root.liege = {
					set_relation_great_misit = root.designated_heir
					if = {
						limit = {
							NOT = { has_variable = misit_liege_power }
						}
						set_variable = {
								name = misit_liege_power
								value = 0
						}
					}
				}
			}
		}
		else_if = {
			limit = { exists = primary_heir }
			primary_heir = {
				if = {
					limit = {
						NOT = { has_variable = misit_power }
					}
					set_variable = {
						name = misit_power
						value = 0
					}
				}
				set_relation_great_misit_liege = root.liege
				add_trait = great_misit
				#great_misit_chronicle_effect = yes
				root.liege = {
					set_relation_great_misit = root.primary_heir
					if = {
						limit = {
							NOT = { has_variable = misit_liege_power }
						}
						set_variable = {
								name = misit_liege_power
								value = 0
						}
					}
				}
				
			}
		}
	}
}
